#########################################Linux command######################################### 

 

### to find out the ip address of a board that sharing internet with it ###

ip neigh show 

 

### check the status of the ethernet connection and make sure the share is enabled ###

nmcli dev status 

### to share internet with other devoce using ethernit  cable ### 

nmcli connection modify "Wired connection 2" ipv4.method shared ipv6.method disabled 
nmcli connection down "Wired connection 2" 
nmcli connection up "Wired connection 2" 

### ssh to a device ### 

ssh <name of the device>@<ip address>

### Create the 500 GB(vertual desk) image file on the SSD ### 

truncate -s 500G "$SSD/variscite_android15_build.img" 
Format that image file as ext4  
mkfs.ext4 -F "$SSD/variscite_android15_build.img" 

 

### This is for the reconfiguring doc . We need this command to mount the ssd ###  

SSD="/media/belal/Extreme Pro" 
IMG="$SSD/variscite_android15_build.img" 
sudo mkdir -p /mnt/var_android15 
LOOPDEV=$(sudo losetup -fP --show "$IMG") 
echo "$LOOPDEV" 
sudo mount -t ext4 "$LOOPDEV" /mnt/var_android15 
sudo chown -R $USER:$USER /mnt/var_android15 

 

### Verify it is been mounted properly and that is the built files are back ###  

ls -ld ~/var_imx-android-15.0.0_1.2.0 
df -h /mnt/var_android15 

 

### Start the docker ###  

sudo systemctl start docker 
sudo systemctl start docker.socket 
To start a new container in the docker you do  
./var-start-container.sh 

 

 

### To UNMOUNT THE SSD FOLLOW STEP BY STEP ### 

sudo systemctl stop docker 
sudo systemctl stop docker.socket 
Sync 
sudo umount /mnt/var_android15  
sudo losetup -d /dev/loop20 
sudo umount "/media/belal/Extreme Pro" 

 


### Rebuilding the android app from the terminal and making sure to use the right Cmake version ###  

cd ~/Downloads/qgroundcontrol 
rm -rf build-android 
export PATH="$HOME/Qt/Tools/CMake/bin:$PATH" 
~/Qt/6.8.3/android_arm64_v8a/bin/qt-cmake \ 
  -S . \ 
  -B build-android \ 
  -G Ninja \ 
  -DCMAKE_BUILD_TYPE=Debug \ 
  -DQT_HOST_PATH="$HOME/Qt/6.8.3/gcc_64" \ 
  -DANDROID_SDK_ROOT="$HOME/Android/Sdk" \ 
  -DANDROID_NDK_ROOT="$HOME/Android/Sdk/ndk/26.1.10909125" \ 
  -DQGC_STABLE_BUILD=ON 
cmake --build build-android --target apk -j"$(nproc)"
